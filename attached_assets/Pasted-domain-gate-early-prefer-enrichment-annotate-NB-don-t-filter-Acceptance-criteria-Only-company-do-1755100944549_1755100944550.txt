// domainRules.ts
export type DomainRules = {
  canonical: string;
  aliases: string[];
  personalBlocks: string[];
};

export function hostToDomain(url = "") {
  try {
    const u = new URL(url.startsWith("http") ? url : `https://${url}`);
    return u.hostname.replace(/^www\./, "");
  } catch {
    return url.replace(/^https?:\/\/(www\.)?/, "").split("/")[0];
  }
}

export function emailDomain(email?: string) {
  return (email || "").toLowerCase().split("@")[1] || "";
}

export function isCompanyDomain(dom: string, rules: DomainRules) {
  const d = dom.toLowerCase();
  return d === rules.canonical || rules.aliases.includes(d);
}

// server/apolloService.ts
function coerceWorkEmailFromSearch(contact: any) {
  return contact.work_email || contact.email || contact.primary_email || null;
}

function shapeFromSearch(raw: any[], rules: DomainRules) {
  const accepted = [] as Array<{
    id: string;
    name: string;
    title?: string;
    email?: string;
    email_domain?: string;
    reasons?: string[];
  }>;
  const skipped = [];

  for (const c of raw) {
    const email = coerceWorkEmailFromSearch(c);
    const dom = emailDomain(email);

    // Reject non-company domains up-front (core contamination fix)
    if (!email || !isCompanyDomain(dom, rules) || rules.personalBlocks.includes(dom)) {
      skipped.push({ c, reason: `domain_mismatch_or_personal:${dom || "none"}` });
      continue;
    }

    accepted.push({
      id: c.id || c.person_id || c.contact_id,
      name: c.name || `${c.first_name ?? ""} ${c.last_name ?? ""}`.trim(),
      title: c.title,
      email,
      email_domain: dom,
    });
  }
  return { accepted, skipped };
}

function pickBestCompanyEmail(enriched: any, rules: DomainRules): string | null {
  const candidates = [
    enriched.work_email,
    enriched.email,
    ...(enriched.emails || []),
  ].filter(Boolean) as string[];

  for (const e of candidates) {
    if (isCompanyDomain(emailDomain(e), rules)) return e;
  }
  return null;
}

async function enrichAndOverrideEmail(shaped: any[], rules: DomainRules) {
  const results = [];
  const skippedEnrichment = [];

  for (const item of shaped) {
    const enriched = await enrichContactByProfile(item); // your existing function
    const override = pickBestCompanyEmail(enriched, rules);

    // prefer enrichment when it finds a better @company email
    const finalEmail = override ?? item.email;

    // final safety: still must be company domain (no NB filtering here)
    if (!finalEmail || !isCompanyDomain(emailDomain(finalEmail), rules)) {
      skippedEnrichment.push({ item, reason: "no_company_email_after_enrichment" });
      continue;
    }

    results.push({ ...item, email: finalEmail });
  }

  return { results, skippedEnrichment };
}

type NBResult = "valid" | "invalid" | "risky" | "catchall" | "unknown";

async function annotateDeliverability(contacts: any[]) {
  const withNB = [];
  const nbErrors = [];

  for (const c of contacts) {
    try {
      const status: NBResult = await validateEmail(c.email); // your NB call
      withNB.push({ ...c, verificationStatus: status, emailVerified: status === "valid" });
    } catch (e) {
      nbErrors.push({ email: c.email, error: String(e) });
      withNB.push({ ...c, verificationStatus: "unknown", emailVerified: false });
    }
  }
  return { withNB, nbErrors };
}
