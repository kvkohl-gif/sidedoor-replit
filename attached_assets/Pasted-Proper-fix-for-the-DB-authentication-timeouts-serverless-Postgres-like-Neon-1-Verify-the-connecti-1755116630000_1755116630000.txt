2) Add safe pool + retry logic (handles “cold start”/sleeping DB)
js
Copy
Edit
// db.js
import pg from 'pg';

const pool = new pg.Pool({
  connectionString: process.env.DATABASE_URL,
  ssl: { rejectUnauthorized: false },
  max: 5,                       // small pool works better on serverless
  idleTimeoutMillis: 30_000,
  connectionTimeoutMillis: 10_000
});

export async function withRetry(fn, { tries = 5, delayMs = 1000 } = {}) {
  let lastErr;
  for (let i = 1; i <= tries; i++) {
    try { return await fn(); }
    catch (err) {
      lastErr = err;
      if (i === tries) break;
      await new Promise(r => setTimeout(r, delayMs * i)); // backoff
    }
  }
  throw lastErr;
}

// Optional: keepalive to prevent idle disconnects
setInterval(() => {
  pool.query('SELECT 1').catch(() => {});
}, 60_000 * 10); // every 10 minutes

export default pool;

4) Use Postgres-backed session store (after DB is reachable)
js
Copy
Edit
import session from 'express-session';
import connectPg from 'connect-pg-simple';
import pool from './db.js';

const PgSession = connectPg(session);

app.use(session({
  store: new PgSession({
    pool,
    tableName: 'session',
    createTableIfMissing: true
  }),
  secret: process.env.SESSION_SECRET,
  resave: false,
  saveUninitialized: false,
  cookie: {
    maxAge: 1000 * 60 * 60 * 24, // 1 day
    sameSite: 'lax',
    secure: false // true if behind HTTPS/Proxy with trust proxy enabled
  }
}));

// If behind a proxy (Replit), also:
app.set('trust proxy', 1);

If it fails, double‑check DATABASE_URL and SESSION_SECRET in Replit Secrets.

6) Common gotchas in Replit
The preview showing “The app is currently not running” usually means the process exited or it isn’t bound to process.env.PORT/0.0.0.0.

Ensure your start script runs the server (e.g., "start": "node index.js" or "node --watch index.js").

Don’t exceed pool size—serverless DBs penalize too many concurrent connections.
