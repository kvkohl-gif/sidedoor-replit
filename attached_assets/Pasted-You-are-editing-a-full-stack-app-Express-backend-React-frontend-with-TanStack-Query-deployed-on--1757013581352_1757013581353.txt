Problem: After OAuth login, the server sets `connect.sid` correctly, but the app returns to the homepage “not logged in”.
Root cause: frontend requests don’t send cookies; and cookie attributes must support Replit’s HTTPS + possible iframe.

Make the following minimal, robust, and reversible changes:

1) Backend — session middleware (server/replitAuth.ts or server entry)
   - Ensure session middleware is registered BEFORE any routes that read auth, and AFTER `app.set('trust proxy', 1)`.
   - Configure the cookie for HTTPS and cross-site safely; do NOT set a `domain`.

   ```ts
   // server/index.ts (or equivalent)
   import session from 'express-session';

   app.set('trust proxy', 1); // required behind Replit proxy

   app.use(session({
     name: 'connect.sid',
     secret: process.env.SESSION_SECRET!,
     resave: false,
     saveUninitialized: false,
     cookie: {
       httpOnly: true,
       secure: true,        // required with SameSite=None on HTTPS
       sameSite: 'none',    // supports Replit iframe/cross-site
       path: '/',
       maxAge: 1000 * 60 * 60 * 24 * 7
       // DO NOT set "domain"
     }
   }));

// client/src/lib/api.ts
export async function api<T = unknown>(path: string, init: RequestInit = {}) {
  const res = await fetch(path, {
    credentials: 'include', // ensure cookie is sent
    headers: { 'content-type': 'application/json', ...(init.headers || {}) },
    ...init,
  });
  if (!res.ok) throw new Error(`${res.status} ${res.statusText}`);
  return res.headers.get('content-type')?.includes('application/json')
    ? await res.json() as T
    : (await res.text() as unknown as T);
}

// client/src/hooks/useAuth.ts
import { useQuery } from '@tanstack/react-query';
import { api } from '../lib/api';

export function useAuth() {
  const { data: user, isLoading } = useQuery({
    queryKey: ['/api/auth/user'],
    queryFn: () => api('/api/auth/user'),
    retry: false,
  });
  return { user, isLoading };
